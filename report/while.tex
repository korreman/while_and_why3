\begin{lstlisting}
 statements = (s ";")+
 s ::= "skip"
     | "assert" c
     | v "<-" e
     | "if" c "then" statements "else" statements "end"
     | "while" c "invariant" i "do" statements "end"
\end{lstlisting}
